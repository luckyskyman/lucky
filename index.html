# models.py from django.db import models class Item(models.Model): name = models.CharField(max_length=255) description = models.TextField() price = models.PositiveIntegerField() seller = models.ForeignKey(User, on_delete=models.CASCADE) sold = models.BooleanField(default=False) # views.py from django.shortcuts import render from .models import Item def item_list(request): items = Item.objects.filter(sold=False) return render(request, 'item_list.html', {'items': items}) def item_detail(request, pk): item = Item.objects.get(pk=pk) return render(request, 'item_detail.html', {'item': item}) def item_new(request): if request.method == 'POST': form = ItemForm(request.POST) if form.is_valid(): item = form.save(commit=False) item.seller = request.user item.save() return redirect('item_detail', pk=item.pk) else: form = ItemForm() return render(request, 'item_new.html', {'form': form}) # urls.py from django.urls import path from .views import item_list, item_detail, item_new urlpatterns = [ path('', item_list, name='item_list'), path('item//', item_detail, name='item_detail'), path('item/new/', item_new, name='item_new'), ]
